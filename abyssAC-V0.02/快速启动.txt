# 1. 安装依赖
pip install -r requirements.txt

# 2. 复制环境变量文件
cp .env.example .env
# 编辑 .env 文件，填入你的API密钥

# 3. 启动系统
python run.py

# 4. 或使用Web API
python run.py --web

配置自定义

# 方式1：修改配置文件
# 编辑 config/config.yaml，然后重启系统

# 方式2：使用API动态修改
from config.config_manager import config_manager

# 更新单个配置项
config_manager.update_config("ai", "model_type", "openai")
config_manager.update_config("kernel", "top_k_nodes", 500)

# 批量更新
config = config_manager.config
config.kernel.top_k_nodes = 500
config.ai.model_type = "openai"
config_manager.save_config(config)
